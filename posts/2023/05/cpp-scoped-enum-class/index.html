#include <iostream>

using namespace std;

// enum - traditional enumeration
enum Week        {Mon   ,Tue   ,Wed  ,Thu   };
enum Season      {Sprint,Summer,Fall ,Winter};

// enum class - scoped enumeratoin
enum class Month {Jan   ,Feb   ,Mar  ,Apr   };
enum class Num   {One   ,Two   ,Three,Four  };

int main() {

  cout << "Is Week a class : " << std::is_class<Week>::value << endl;
  cout << "Is Week a enum : " << std::is_enum<Week>::value << endl;
  cout << "Is Month a class : " << std::is_class<Month>::value << endl;
  cout << "Is Month a enum : " << std::is_enum<Month>::value << endl << endl;

  Week w = Tue;
  Season s = Summer;
  Month m = Month::Feb;
  Num n = Num::Two;
  Num n2 = Num::Two;

  if (w == s) {
    cout << "w == s, ohh!" << endl;  // unmeaningful
